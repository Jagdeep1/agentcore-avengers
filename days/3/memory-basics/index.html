Memory Basics

Remember Day 1? Your agent forgot everything between requests. That’s because AI models are inherently stateless - each request starts fresh with no knowledge of previous interactions.

AgentCore Memory solves this by providing managed memory services that persist context across interactions.

Two Types of Memory

AgentCore Memory provides two complementary memory types:

┌─────────────────────────────────────────────────────────────────────┐
│                        Memory Types                                  │
├─────────────────────────────────────────────────────────────────────┤
│                                                                      │
│  SHORT-TERM MEMORY                    LONG-TERM MEMORY               │
│  ════════════════                     ═══════════════                │
│                                                                      │
│  Scope: Single session                Scope: Across all sessions     │
│  Duration: Session lifetime           Duration: Persistent           │
│  Content: Raw conversation turns      Content: Extracted insights    │
│                                                                      │
│  ┌─────────────────────┐             ┌─────────────────────┐        │
│  │ User: My name is    │             │ Preferences:        │        │
│  │       Sarah         │             │ - Prefers Python    │        │
│  │                     │             │ - Morning meetings  │        │
│  │ Agent: Hello Sarah! │             │                     │        │
│  │                     │             │ Facts:              │        │
│  │ User: I prefer      │      ──►    │ - Name: Sarah       │        │
│  │       Python        │             │ - Team: Engineering │        │
│  │                     │             │                     │        │
│  │ Agent: Noted!       │             │ Summary:            │        │
│  └─────────────────────┘             │ - Discussed project │        │
│                                      │   setup and prefs   │        │
│                                      └─────────────────────┘        │
│                                                                      │
└─────────────────────────────────────────────────────────────────────┘

When to Use Each

Short-term: For immediate context like “what did I just ask?” Long-term: For persistent knowledge like “what are my preferences?”

Creating a Memory Resource

Step 1: Create Memory via SDK

import boto3

memory_client = boto3.client('bedrock-agentcore-memory', region_name='us-east-1')

# Create a memory resource
response = memory_client.create_memory(
    name='day3-agent-memory',
    description='Memory for Day 3 agent with conversation persistence',

    # Enable both memory types
    memoryConfig={
        'shortTermMemory': {
            'enabled': True,
            'maxTurns': 50,  # Keep last 50 conversation turns
        },
        'longTermMemory': {
            'enabled': True,
            'strategies': ['USER_PREFERENCE', 'SEMANTIC', 'SUMMARY']
        }
    },

    tags={
        'project': 'agentcore-initiative',
        'day': '3'
    }
)

memory_id = response['memoryId']
print(f"Created memory: {memory_id}")

Step 2: Configure Memory Settings

# Update memory configuration
memory_client.update_memory(
    memoryId=memory_id,
    memoryConfig={
        'shortTermMemory': {
            'enabled': True,
            'maxTurns': 100,
            'tokenLimit': 8000,  # Max tokens for context window
        },
        'longTermMemory': {
            'enabled': True,
            'strategies': ['USER_PREFERENCE', 'SEMANTIC', 'SUMMARY'],
            'consolidation': {
                'enabled': True,
                'interval': 'END_OF_SESSION'  # When to consolidate
            }
        }
    }
)

Integrating Memory with Strands

Basic Integration

# agent_with_memory.py
from strands import Agent
from strands.models import BedrockModel
from bedrock_agentcore.memory import MemoryClient
from bedrock_agentcore.runtime import runtime_handler

# Initialize clients
model = BedrockModel(model_id="anthropic.claude-sonnet-4-20250514-v1:0")
memory = MemoryClient(memory_id='mem-xxxxxxxxxxxx')

# Create agent
agent = Agent(
    model=model,
    system_prompt="""You are a helpful assistant with memory capabilities.
You can remember previous conversations and user preferences.
Use your memory to provide personalized, contextual responses."""
)

@runtime_handler
def handler(event, context):
    session_id = event.get('session_id', 'default')
    user_id = event.get('userContext', {}).get('sub', 'anonymous')
    message = event.get('message', '')

    # Retrieve conversation history (short-term)
    history = memory.get_session_history(
        sessionId=session_id,
        maxTurns=20
    )

    # Retrieve relevant long-term memories
    relevant_memories = memory.search_memories(
        userId=user_id,
        query=message,
        maxResults=5
    )

    # Build context for the agent
    context_prompt = build_context(history, relevant_memories)

    # Get response with context
    response = agent(f"{context_prompt}\n\nUser: {message}")

    # Save the interaction to memory
    memory.add_message(
        sessionId=session_id,
        role='user',
        content=message
    )
    memory.add_message(
        sessionId=session_id,
        role='assistant',
        content=str(response)
    )

    return {
        'response': str(response),
        'session_id': session_id
    }

def build_context(history, memories):
    """Build context string from history and memories."""
    context_parts = []

    if memories:
        context_parts.append("Relevant memories about this user:")
        for mem in memories:
            context_parts.append(f"- {mem['content']}")

    if history:
        context_parts.append("\nRecent conversation:")
        for turn in history[-10:]:  # Last 10 turns
            context_parts.append(f"{turn['role'].title()}: {turn['content']}")

    return '\n'.join(context_parts)

Using the Strands Memory Integration

Strands Agents SDK provides built-in memory support:

# Using Strands built-in memory integration
from strands import Agent
from strands.models import BedrockModel
from strands.memory import AgentCoreMemory

# Initialize memory with AgentCore
memory = AgentCoreMemory(
    memory_id='mem-xxxxxxxxxxxx',
    region='us-east-1'
)

# Create agent with automatic memory management
agent = Agent(
    model=BedrockModel(model_id="anthropic.claude-sonnet-4-20250514-v1:0"),
    memory=memory,  # Strands handles memory automatically
    system_prompt="You are a helpful assistant that remembers user preferences."
)

# Memory is automatically managed!
response = agent("My name is Sarah and I prefer Python.")
print(response)

# Later in another session...
response = agent("What's my name and preferred language?")
print(response)  # "Your name is Sarah and you prefer Python!"

Testing Memory Persistence

Test Short-Term Memory

# test_short_term.py
def test_short_term_memory():
    """Test that agent remembers within a session."""

    session_id = "test-session-001"

    # Turn 1: Introduce yourself
    response1 = invoke_agent(
        message="Hi, my name is Alex and I'm working on a data pipeline.",
        session_id=session_id
    )
    print(f"Turn 1: {response1}")

    # Turn 2: Reference previous context
    response2 = invoke_agent(
        message="What was I working on again?",
        session_id=session_id
    )
    print(f"Turn 2: {response2}")
    # Expected: "You mentioned you're working on a data pipeline."

    # Turn 3: Continue the conversation
    response3 = invoke_agent(
        message="Can you help me with that?",
        session_id=session_id
    )
    print(f"Turn 3: {response3}")
    # Agent should remember the data pipeline context

test_short_term_memory()

Test Long-Term Memory

# test_long_term.py
def test_long_term_memory():
    """Test that agent remembers across sessions."""

    user_id = "user-123"

    # Session 1: Establish preferences
    session1 = "session-001"
    invoke_agent(
        message="I always prefer TypeScript over JavaScript.",
        session_id=session1,
        user_id=user_id
    )
    invoke_agent(
        message="My favorite color is blue.",
        session_id=session1,
        user_id=user_id
    )

    # End session 1 (triggers memory consolidation)
    end_session(session1)

    # Session 2: New session, same user
    session2 = "session-002"
    response = invoke_agent(
        message="If I'm starting a new project, what language should I use?",
        session_id=session2,
        user_id=user_id
    )
    print(f"Session 2: {response}")
    # Expected: Mentions TypeScript based on remembered preference

test_long_term_memory()

Memory Working!

If your agent remembers the user’s preference from a previous session, your long-term memory is correctly configured!

Memory API Reference

Core Operations

# Add a message to session history
memory.add_message(
    sessionId='session-123',
    role='user',  # or 'assistant'
    content='Hello!'
)

# Get session history
history = memory.get_session_history(
    sessionId='session-123',
    maxTurns=50
)

# Search long-term memories
memories = memory.search_memories(
    userId='user-456',
    query='What are my preferences?',
    maxResults=10
)

# Manually add a long-term memory
memory.add_memory(
    userId='user-456',
    content='Prefers morning meetings',
    memoryType='USER_PREFERENCE'
)

# Delete a memory
memory.delete_memory(memoryId='memory-789')

# End session (triggers consolidation)
memory.end_session(sessionId='session-123')

Session Management

# Create a new session
session = memory.create_session(
    userId='user-456',
    metadata={
        'source': 'web-app',
        'topic': 'project-planning'
    }
)

# Get session info
session_info = memory.get_session(sessionId='session-123')
print(f"Session started: {session_info['createdAt']}")
print(f"Turn count: {session_info['turnCount']}")

# List user sessions
sessions = memory.list_sessions(userId='user-456')
for s in sessions:
    print(f"Session {s['sessionId']}: {s['turnCount']} turns")
